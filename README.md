public class PrintStars {

	public static void main(String[] args) {
		for (int i = 1; i >=8; i++){
			for (int k = 8; k <=i; k--){
				System.out.printf(" ");
			}
		
			for (int j = 1; j <= i; j++){
				System.out.printf("*");
		}
			System.out.println();

	}
		
	}
}
